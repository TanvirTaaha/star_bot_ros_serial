long mapp(long x, long in_min, long in_max, long out_min, long out_max)
{
  return (x - in_min) * (out_max - out_min) / (in_max - in_min) + out_min;
}

int *any(char *str)
{
  static int angles[4];
  sscanf(str, "%d,%d,%d", &angles[0], &angles[1], &angles[2]);
  return angles;
}
